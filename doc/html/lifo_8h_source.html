165  void * reallocate(void * i_block, size_t i_old_size, size_t i_new_size)
166  {
167  // branch depending on the old block: this check detects page switches and external blocks
168  if (same_page(i_block, reinterpret_cast<void*>(m_top)))
169  {
170  // the following call sets the top only when it is sure to not throw
171  auto const new_block = set_top_and_allocate(i_block, i_new_size);
172 
173  copy(i_block, i_old_size, new_block, i_new_size);
174  return new_block;
175  }
176  else
177  {
178  if (i_old_size < UNDERLYING_ALLOCATOR::page_size)
179  {
180  auto const page_to_deallocate = reinterpret_cast<void*>(m_top);
181  DENSITY_ASSERT_INTERNAL(!same_page(page_to_deallocate, i_block));
182 
183  // the following call sets the top only when it is sure to not throw
184  auto const new_block = set_top_and_allocate(i_block, i_new_size);
185 
186  copy(i_block, i_old_size, new_block, i_new_size);
187 
188  UNDERLYING_ALLOCATOR::deallocate_page(page_to_deallocate);
189  return new_block;

191  else

193  // the old block is external

195  copy(i_block, i_old_size, new_block, i_new_size);
196  UNDERLYING_ALLOCATOR::deallocate(i_block, i_old_size, alignment);
197  return new_block;
198  }

280  void * set_top_and_allocate(void * const i_current_top, size_t const i_size)
281  {
282  DENSITY_ASSERT_INTERNAL(i_size % alignment == 0);
283 
284  auto const current_top = reinterpret_cast<uintptr_t>(i_current_top);
285 
286  // we have to procrastinate the assignment of m_top until we have allocated the page or the external block
287  auto const new_top = current_top + i_size;
288  auto const new_offset = new_top - uint_lower_align(current_top, UNDERLYING_ALLOCATOR::page_alignment);
289  if (new_offset < UNDERLYING_ALLOCATOR::page_size)
290  {
291  DENSITY_ASSERT_INTERNAL(i_size <= UNDERLYING_ALLOCATOR::page_size);
292  auto const new_block = i_current_top;
293  m_top = new_top;
294  return new_block;
295  }
296  else if (i_size < UNDERLYING_ALLOCATOR::page_size)
297  {
298  // allocate a new page
299  auto const new_page = UNDERLYING_ALLOCATOR::allocate_page();
300  m_top = reinterpret_cast<uintptr_t>(new_page) + i_size;
301  return new_page;
302  }
303  else
304  {
305  // external block
306  auto const new_external_block = UNDERLYING_ALLOCATOR::allocate(i_size, alignment);
307  m_top = reinterpret_cast<uintptr_t>(i_current_top);
308  return new_external_block;

618  template <typename... CONSTRUCTION_PARAMS>
619  lifo_array(size_t i_size, const CONSTRUCTION_PARAMS &... i_construction_params )
620  : detail::LifoArrayImpl<TYPE>(i_size)
621  {
622  size_t element_index = 0;
623  try
624  {
625  for (; element_index < i_size; element_index++)
626  {
627  new(m_elements + element_index) TYPE(i_construction_params...);
628  }
629  }
630  catch (...)
631  {
632  // destroy all the elements constructed till now, and then deallocate
633  auto it = m_elements + element_index;
634  it--;
635  for (; it >= m_elements; it--)
636  {
637  it->TYPE::~TYPE();
638  }
639  throw;
