169  void * reallocate(void * i_block, size_t i_old_size, size_t i_new_size)
170  {
171  // branch depending on the old block: this check detects page switches and external blocks
172  if (same_page(i_block, reinterpret_cast<void*>(m_top)))
173  {
174  // the following call sets the top only when it is sure to not throw
175  auto const new_block = set_top_and_allocate(i_block, i_new_size);
176 
177  copy(i_block, i_old_size, new_block, i_new_size);
178  return new_block;
179  }
180  else
181  {
182  if (i_old_size < UNDERLYING_ALLOCATOR::page_size)
183  {
184  auto const page_to_deallocate = reinterpret_cast<void*>(m_top);
185  DENSITY_ASSERT_INTERNAL(!same_page(page_to_deallocate, i_block));
186 
187  // the following call sets the top only when it is sure to not throw
188  auto const new_block = set_top_and_allocate(i_block, i_new_size);
189 
190  copy(i_block, i_old_size, new_block, i_new_size);
191 
192  UNDERLYING_ALLOCATOR::deallocate_page(page_to_deallocate);
193  return new_block;
194  }
195  else
196  {
197  // the old block is external
198  auto const new_block = allocate(i_new_size);
199  copy(i_block, i_old_size, new_block, i_new_size);
200  UNDERLYING_ALLOCATOR::deallocate(i_block, i_old_size, alignment);

252  void * set_top_and_allocate(void * const i_current_top, size_t const i_size)
253  {
254  DENSITY_ASSERT_INTERNAL(i_size % alignment == 0);
255 
256  auto const current_top = reinterpret_cast<uintptr_t>(i_current_top);
257 
258  // we have to procrastinate the assignment of m_top until we have allocated the page or the external block
259  auto const new_top = current_top + i_size;
260  auto const new_offset = new_top - uint_lower_align(current_top, UNDERLYING_ALLOCATOR::page_alignment);
261  if (new_offset < UNDERLYING_ALLOCATOR::page_size)
262  {
263  DENSITY_ASSERT_INTERNAL(i_size <= UNDERLYING_ALLOCATOR::page_size);
264  auto const new_block = i_current_top;
265  m_top = new_top;
266  return new_block;
267  }
268  else
269  {
270  // this branch does not use the value of m_top
271  return allocate_slow_path(i_size);
272  }
273  }
