428  inline void * aligned_allocate(size_t i_size, size_t i_alignment, size_t i_alignment_offset = 0)
429  {
430  DENSITY_ASSERT(i_alignment > 0 && is_power_of_2(i_alignment));
431  DENSITY_ASSERT(i_alignment_offset <= i_size);
432 
433  void * user_block;
434  if (i_alignment <= detail::MaxAlignment && i_alignment_offset == 0)
435  {
436  user_block = operator new(i_size);
437  }
438  else
439  {
440  // reserve an additional space in the block equal to the max(i_alignment, sizeof(AlignmentHeader) - sizeof(void*) )
441  size_t const extra_size =
442  detail::size_max(i_alignment, sizeof(detail::AlignmentHeader));
443  size_t const actual_size = i_size + extra_size;
444  auto const complete_block = operator new(actual_size);
445  user_block = address_lower_align(
446  address_add(complete_block, extra_size), i_alignment, i_alignment_offset);
447  detail::AlignmentHeader & header =
448  *(static_cast<detail::AlignmentHeader *>(user_block) - 1);
449  header.m_block = complete_block;
450  DENSITY_ASSERT_INTERNAL(
451  user_block >= complete_block &&
452  address_add(user_block, i_size) <= address_add(complete_block, actual_size));
453  }
454  return user_block;
455  }
